package com.example.alex.checkboxcolor;

import android.graphics.Color;
import android.support.v7.app.ActionBarActivity;
import android.os.Bundle;
import android.view.Gravity;
import android.view.Menu;
import android.view.MenuItem;
import android.view.View;
import android.view.ViewGroup;
import android.widget.AbsoluteLayout;
import android.widget.CheckBox;
import android.widget.CompoundButton;
import android.widget.GridLayout;
import android.widget.LinearLayout;
import android.widget.TextView;
import android.widget.Toast;

public class MainActivity extends ActionBarActivity {

    private GridLayout mainLL;

    private TextView tv;
    private CheckBox cbR;
    private CheckBox cbG;
    private CheckBox cbB;

    private byte red;
    private byte green;
    private byte blue;


    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        this.cbR    = (CheckBox)this.findViewById(R.id.red);
        this.cbG    = (CheckBox)this.findViewById(R.id.green);
        this.cbB    = (CheckBox)this.findViewById(R.id.blue);
        this.tv     = (TextView)this.findViewById(R.id.tv);
        this.mainLL = (GridLayout)this.findViewById(R.id.mainLL);

        if(cbR.isChecked())
        {
            this.red = 1;
        }
        else
        {
            this.red = 0;
        }

        if(cbG.isChecked())
        {
            this.green = 1;
        }
        else
        {
            this.green = 0;
        }

        if(cbB.isChecked())
        {
            this.blue = 1;
        }
        else
        {
            this.blue = 0;
        }
    }

    public void checkBoxClick(View v)
    {
        //CheckBox cb = (CheckBox)this.findViewById(v.getId());
        if(v.getId() == R.id.red)
        {
            Toast.makeText(this,"Red Enable",Toast.LENGTH_SHORT).show();
        }
        else
        if(v.getId() == R.id.green)
        {
            Toast.makeText(this,"green Enable",Toast.LENGTH_SHORT).show();
        }
        else
        if(v.getId() == R.id.blue)
        {
            Toast.makeText(this,"Blue Enable",Toast.LENGTH_SHORT).show();
        }
        else
        {
            Toast.makeText(this,"Nothing",Toast.LENGTH_SHORT).show();
        }


        if(this.cbR.isChecked() & this.cbG.isChecked() & this.cbB.isChecked())
        {
            this.mainLL.setBackgroundColor(Color.rgb(255,255,255));
            this.tv.setTextColor(Color.rgb(0, 0, 0));
        }
        else
        if(this.cbG.isChecked() & this.cbB.isChecked())
        {
            this.mainLL.setBackgroundColor(Color.rgb(0,255,255));
            this.tv.setTextColor(Color.rgb(255, 0, 0));
        }
        else
        if(this.cbR.isChecked() && this.cbG.isChecked())
        {
            this.mainLL.setBackgroundColor(Color.rgb(255,255,0));
            this.tv.setTextColor(Color.rgb(0,0,255));
        }
        else
        if(this.cbR.isChecked() && this.cbB.isChecked())
        {
            this.mainLL.setBackgroundColor(Color.rgb(255,0,255));
            this.tv.setTextColor(Color.rgb(0,255,0));
        }
        else
        if(this.cbR.isChecked())
        {
            this.mainLL.setBackgroundColor(Color.rgb(255,0,0));
            this.tv.setTextColor(Color.rgb(0, 255, 255));
        }
        else
        if(this.cbG.isChecked())
        {
            this.mainLL.setBackgroundColor(Color.rgb(0,255,0));
            this.tv.setTextColor(Color.rgb(255,0,255));
        }
        else
        if(this.cbB.isChecked())
        {
            this.mainLL.setBackgroundColor(Color.rgb(0,0,255));
            this.tv.setTextColor(Color.rgb(255, 255, 0));
        }
        else
        {
            this.mainLL.setBackgroundColor(Color.rgb(0,0,0));
            this.tv.setTextColor(Color.rgb(255,255,255));
        }
    }
}
